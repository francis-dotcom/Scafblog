```jsx
// src/components/Newsletter.js
import React, { useState } from "react";

export default function Newsletter({
  title = "üìß Subscribe to Our Newsletter",
  description = "Get the latest posts delivered to your inbox",
  buttonText = "Subscribe",
}) {
  const [email, setEmail] = useState("");
  const [status, setStatus] = useState("");
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e) => {
    e.preventDefault();
    if (!email || !email.includes("@")) {
      setStatus("error");
      return;
    }

    setLoading(true);
    setStatus("");

    try {
      const response = await fetch("/.netlify/functions/subscribe", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ email }),
      });

      if (response.ok) {
        setStatus("success");
        setEmail("");
      } else {
        setStatus("error");
      }
    } catch (error) {
      setStatus("error");
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="newsletter">
      <h3>{title}</h3>
      <p>{description}</p>

      <form onSubmit={handleSubmit}>
        <div className="input-group">
          <input
            type="email"
            value={email}
            onChange={(e) => setEmail(e.target.value)}
            placeholder="Enter your email"
            required
            disabled={loading}
          />
          <button type="submit" disabled={loading || !email}>
            {loading ? "‚è≥ Subscribing..." : `‚ú® ${buttonText}`}
          </button>
        </div>
      </form>

      {status === "success" && (
        <div className="success">üéâ Successfully subscribed!</div>
      )}

      {status === "error" && (
        <div className="error">‚ùå Something went wrong. Try again.</div>
      )}
    </div>
  );
}
```

### Vanilla JavaScript

```javascript
class Newsletter {
  constructor(containerId, options = {}) {
    this.container = document.getElementById(containerId);
    this.options = {
      title: "üìß Subscribe to Our Newsletter",
      description: "Get the latest posts delivered to your inbox",
      buttonText: "Subscribe",
      ...options,
    };
    this.render();
    this.bindEvents();
  }

  render() {
    this.container.innerHTML = `
      <div class="newsletter">
        <h3>${this.options.title}</h3>
        <p>${this.options.description}</p>
        
        <form class="newsletter-form">
          <div class="input-group">
            <input type="email" placeholder="Enter your email" required />
            <button type="submit">‚ú® ${this.options.buttonText}</button>
          </div>
        </form>
        
        <div class="message"></div>
      </div>
    `;
  }

  bindEvents() {
    const form = this.container.querySelector(".newsletter-form");
    const input = this.container.querySelector("input");
    const button = this.container.querySelector("button");
    const message = this.container.querySelector(".message");

    form.addEventListener("submit", async (e) => {
      e.preventDefault();

      if (!input.value?.includes("@")) {
        this.showMessage("error", "‚ùå Please enter a valid email");
        return;
      }

      button.disabled = true;
      button.textContent = "‚è≥ Subscribing...";

      try {
        const response = await fetch("/.netlify/functions/subscribe", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ email: input.value }),
        });

        if (response.ok) {
          this.showMessage("success", "üéâ Successfully subscribed!");
          input.value = "";
        } else {
          this.showMessage("error", "‚ùå Something went wrong");
        }
      } catch {
        this.showMessage("error", "‚ùå Network error");
      } finally {
        button.disabled = false;
        button.textContent = `‚ú® ${this.options.buttonText}`;
      }
    });
  }

  showMessage(type, text) {
    const message = this.container.querySelector(".message");
    message.className = `message ${type}`;
    message.textContent = text;
    setTimeout(() => (message.textContent = ""), 5000);
  }
}

// Usage: new Newsletter('newsletter-container');
```

## Step 4: Backend Function

### For Netlify

Create `netlify/functions/subscribe.js`:

```javascript
exports.handler = async (event, context) => {
  if (event.httpMethod !== "POST") {
    return {
      statusCode: 405,
      headers: { "Access-Control-Allow-Origin": "*" },
      body: JSON.stringify({ message: "Method Not Allowed" }),
    };
  }

  try {
    const { email } = JSON.parse(event.body);

    if (!email || !email.includes("@")) {
      return {
        statusCode: 400,
        headers: { "Access-Control-Allow-Origin": "*" },
        body: JSON.stringify({ message: "Invalid email" }),
      };
    }

    const response = await fetch(
      `https://emailoctopus.com/api/1.6/lists/${process.env.EMAILOCTOPUS_LIST_ID}/contacts`,
      {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          api_key: process.env.EMAILOCTOPUS_API_KEY,
          email_address: email,
          status: "SUBSCRIBED",
        }),
      }
    );

    const data = await response.json();

    if (response.ok) {
      return {
        statusCode: 200,
        headers: { "Access-Control-Allow-Origin": "*" },
        body: JSON.stringify({ message: "Success!" }),
      };
    }

    // Handle duplicate emails gracefully
    if (data.error?.code === "MEMBER_EXISTS_WITH_EMAIL_ADDRESS") {
      return {
        statusCode: 400,
        headers: { "Access-Control-Allow-Origin": "*" },
        body: JSON.stringify({ message: "Already subscribed!" }),
      };
    }

    return {
      statusCode: 400,
      headers: { "Access-Control-Allow-Origin": "*" },
      body: JSON.stringify({ message: "Subscription failed" }),
    };
  } catch (error) {
    return {
      statusCode: 500,
      headers: { "Access-Control-Allow-Origin": "*" },
      body: JSON.stringify({ message: "Server error" }),
    };
  }
};
```

```javascript
export default async function handler(req, res) {

    return res.status(405).json({ message: "Method Not Allowed" });
  }

  try {
    const { email } = req.body;

    if (!email || !email.includes("@")) {
      return res.status(400).json({ message: "Invalid email" });
    }

    const response = await fetch(
      `https://emailoctopus.com/api/1.6/lists/${process.env.EMAILOCTOPUS_LIST_ID}/contacts`,
      {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          api_key: process.env.EMAILOCTOPUS_API_KEY,
          email_address: email,
          status: "SUBSCRIBED",
        }),
      }
    );

    const data = await response.json();

    if (response.ok) {
      return res.status(200).json({ message: "Success!" });
```
